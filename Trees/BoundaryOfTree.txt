def left_print(root, result):
    if root:
        if root.left:
            result.append(str(root.val))
            left_print(root.left, result)
        elif root.right:
            result.append(str(root.val))
            left_print(root.right, result)

def right_print(root, result):
    if root:
        if root.right:
            right_print(root.right, result)
            result.append(str(root.val))
        elif root.left:
            right_print(root.left, result)
            result.append(str(root.val))
