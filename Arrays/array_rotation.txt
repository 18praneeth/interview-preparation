# 4, 5, 6, 1, 2, 3

def rotaion_of_array(arr):
    l = 0
    h = len(arr) - 1

    while(l <= h): # l = low, h = high
        mid = l + (h-l)//2
        
        if(mid+1 < mid): #  can add mid < high and
            return mid + 1
        if(mid < mid-1): # low < mid
            return mid
        
        if(arr[mid] < arr[h]):
            h = mid - 1
        else:
            l = mid + 1

    return 0;
